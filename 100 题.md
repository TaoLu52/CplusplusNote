#### [1056. 易混淆数](https://leetcode.cn/problems/confusing-number/)
```c++
class Solution {

public:

    bool confusingNumber(int n) {

        vector<int> vec;

        int temp = n;

        if(n==0 || n==1 || n==8)

        {

            return false;

        }

        while(temp>0)

        {

            int r = temp%10;

            temp=temp/10;

            if(!isconfuis(r))

            {

                return false;

            }

            vec.push_back(r);      

        }

        if(vec.size()==1)

        {

            return true;

        }

        int m=1;

        int n_s=0;

        int cnt=0;

        for(cnt=0;cnt<vec.size();cnt++)

        {

            int x=retconfuis(vec[vec.size()-cnt-1]);

            n_s+=x*m;

            m=m*10;

        }

        if(n_s==n)

        {

            return false;

        }

        return true;

    }

    bool isconfuis(int n)

    {

        switch(n)

        {

            case 0:

            case 1:

            case 6:

            case 8:

            case 9:

                return true;

            case 2:

            case 3:

            case 4:

            case 5:

            case 7:

                return false;

        }

        return true;

    }

    int retconfuis(int n)

    {

        switch(n)

        {

            case 0:

                return 0;

            case 1:

                return 1;

            case 6:

                return 9;

            case 8:

                return 8;

            case 9:

                return 6;

        }

        return -1;

    }

};
```
